import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

public class HistoricoCompras {
    public static class Compra {
        private LocalDate data;
        private List<String> itensComprados;
        private double valorTotal;

        public Compra(LocalDate data, List<String> itensComprados, double valorTotal) {
            this.data = data;
            this.itensComprados = itensComprados;
            this.valorTotal = valorTotal;
        }

        @Override
        public String toString() {
            return "Data: " + data + ", Itens: " + itensComprados + ", Valor Total: R$" + valorTotal;
        }
    }

    private List<Compra> compras;

    public HistoricoCompras() {
        this.compras = new ArrayList<>();
    }

    public void adicionarCompra(Compra compra) {
        compras.add(compra);
    }

    public void listarCompras() {
        if (compras.isEmpty()) {
            System.out.println("Nenhum hist√≥rico de compra.");
        } else {
            for (Compra c : compras) {
                System.out.println(c);
            }
        }
    }
}
