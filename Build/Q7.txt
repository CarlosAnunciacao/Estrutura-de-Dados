import java.time.LocalDate;

public class Pet {
    private String nome;
    private String telefoneTutor;
    private LocalDate dataNascimento;
    private String raca;

    public Pet(String nome, String telefoneTutor, LocalDate dataNascimento, String raca) {
        this.nome = nome;
        this.telefoneTutor = telefoneTutor;
        this.dataNascimento = dataNascimento;
        this.raca = raca;
    }

    public String getNome() {
        return nome;
    }

    public String getTelefoneTutor() {
        return telefoneTutor;
    }

    public LocalDate getDataNascimento() {
        return dataNascimento;
    }

    public String getRaca() {
        return raca;
    }

    public void setTelefoneTutor(String telefoneTutor) {
        this.telefoneTutor = telefoneTutor;
    }
}
