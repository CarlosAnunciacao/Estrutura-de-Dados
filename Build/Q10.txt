import java.util.ArrayList;
import java.util.List;

public class Playlist {
    public static class Musica {
        private String titulo;
        private String artista;
        private int duracaoEmSegundos;

        public Musica(String titulo, String artista, int duracaoEmSegundos) {
            this.titulo = titulo;
            this.artista = artista;
            this.duracaoEmSegundos = duracaoEmSegundos;
        }

        @Override
        public String toString() {
            return titulo + " - " + artista + " (" + duracaoEmSegundos + "s)";
        }
    }

    private List<Musica> musicas;

    public Playlist() {
        this.musicas = new ArrayList<>();
    }

    public void adicionarMusica(Musica musica) {
        musicas.add(musica);
    }

    public boolean removerMusica(String titulo) {
        return musicas.removeIf(m -> m.titulo.equals(titulo));
    }

    public void listarMusicas() {
        if (musicas.isEmpty()) {
            System.out.println("A playlist est√° vazia.");
        } else {
            for (Musica musica : musicas) {
                System.out.println(musica);
            }
        }
    }
}
